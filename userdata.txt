#!/bin/bash
yum update -y
yum install -y httpd24 php56
yum install -y php56-devel php56-mysqlnd php56-pdo php56-mbstring
yum install -y stress
cd /etc/httpd/conf
cp httpd.conf httpdconfbackup.conf
rm -rf httpd.conf
wget https://raw.githubusercontent.com/LewisLebentz/Website/master/httpd.conf
cd /var/www/html
aws s3 sync --delete s3://lew-wordpress-code/ /var/www/html/
chmod -R 755 wp-content
chown -R apache.apache wp-content
wget https://raw.githubusercontent.com/LewisLebentz/Website/master/htaccess
mv htaccess .htaccess
cd /etc
rm -rf crontab
wget https://raw.githubusercontent.com/LewisLebentz/Website/master/crontab
service httpd start
chkconfig httpd on
